###### 3.张量的模n分解和合成MATLAB代码

模N分解：

```matlab
%model n 分解
function matrix = unfolding(tensor,n1,n2,n3,mode)
if mode == 1
    matrix = zeros(n1,n2*n3);
    for k = 1:n3
        matrix(:,(k-1)*n2+1:k*n2) = reshape(tensor(:,:,k),n1,n2); % mode-1 unfolding
    end
elseif mode == 2
    matrix = zeros(n2,n1*n3);
    for k = 1:n3
        matrix(:,(k-1)*n1+1:k*n1) = reshape(tensor(:,:,k),n1,n2)'; % mode-2 unfolding
    end
else
    matrix = zeros(n3,n1*n2);
    for j = 1:n2
        matrix(:,(j-1)*n1+1:j*n1) = reshape(tensor(:,j,:),n1,n3)'; % mode-3 unfolding
    end
end
```

  模N合成

```matlab
%model n 合成
function tensor = folding(matrix,n1,n2,n3,mode)
tensor = zeros(n1,n2,n3);
if mode == 1
    for k = 1:n3
        tensor(:,:,k) = matrix(:,(k-1)*n2+1:k*n2); % mode-1 folding
    end
elseif mode == 2
    for k = 1:n3
        tensor(:,:,k) = matrix(:,(k-1)*n1+1:k*n1)'; % mode-2 folding
    end
else
    for j = 1:n2
        tensor(:,j,:) = matrix(:,(j-1)*n1+1:j*n1)'; % mode-3 folding
    end
end
```

###### Python排序代码

```python

# InsertSort
array = [8,3,1,5,2,1]

def InsertSort(Num):
    for i in  range(1,len(Num)):
        if Num[i-1]>Num[i]:
            temp = Num[i]
            index = i
            while index>0 and Num[index-1]>temp:
                Num[index-1],Num[index] = Num[index],Num[index-1]
                index-=1

def main():
    InsertSort(array)
    print(array)

main()

# sum(1+2+3+....+n) = O(n^2)
```
